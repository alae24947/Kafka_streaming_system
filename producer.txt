INITIALIZE Kafka producer with broker address
DEFINE list of products

INITIALIZE counters for validation:
    total_sent ← 0
    total_filtered ← 0

WHILE true DO
    GENERATE random order:
        order_id
        user_id
        product
        quantity
        price
        timestamp

    SERIALIZE order into JSON format
    SEND order to Kafka topic "orders"

    INCREMENT total_sent
    IF price > 1000 THEN
        INCREMENT total_filtered
    END IF

    DISPLAY sent order and statistics
    WAIT for 1 second
END WHILE
